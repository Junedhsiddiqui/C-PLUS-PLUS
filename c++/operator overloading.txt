
//operator overloadig
//sum of two numbers..
#include<iostream>
using namespace std;

class Complex
{
    private:
       int a,b;
    public:
       void setdata(int x)
       {
           a=x;
       }
       void showdata()
       {
           cout<<"a: "<<a;
        }
    Complex operator +(Complex c)
     {
         Complex temp;
         temp.a=a+c.a;
         return (temp);
     }
};
int main()
{
    Complex c1,c2,c3;
    c1.setdata(4);
    c2.setdata(4);
    c3=c1+c2;
    c3.showdata();
}
