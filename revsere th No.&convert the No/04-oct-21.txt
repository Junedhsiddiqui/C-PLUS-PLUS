
//Q.Reverse the Number..
#include<stdio.h>
int main()
{
    int num,rem,rev=0;
    scanf("%d",&num);
    while(num!=0)
    {
        rem=num%10;
        rev=(rev*10)+rem;
        num=num/10;
    }
    printf("%d",rev);
    return 0;
    
}


//Q.Convert the No. Decimal to Binary..
#include<stdio.h>
int main()
{
    int num,rem,bin=0;
    scanf("%d",&num);
    while(num!=0)
    {
        rem=num%2;
        num=num/2;
        bin=(bin*10)+rem;
        
    }
    printf("%d",bin);
    return 0;
    
}
