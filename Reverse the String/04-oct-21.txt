//Q.Reverse the string..
#include<stdio.h>
#include<string.h>
int main()
{
    char str[200],len,temp,i;
    printf("Enter the string:");
    fgets(str,200,stdin);
    len=strlen(str);
    for(i=0;i<len/2;i++)
    {
        temp=str[i];
        str[i]=str[len-i-1];
        str[len-i-1]=temp;
    }
    printf("%s",str);
    return 0;
    
}
